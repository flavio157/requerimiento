create view V_MAT_SIN_DEVOLVER
as
SELECT ms.PERSONAL_SOLICITO as CODIGO_PER, ms.FECH_REGISTRO as FECHA , ds.COD_PRODUCTO as CODIGO
,ds.NUM_SERIE as SERIE ,p.DES_PRODUCTO as MATERIAL from T_MATERIALES_SALIDA as ms
inner join T_DETSALIDA as ds
on ms.CODIGO_SALIDA = ds.CODIGO_SALIDA
inner join T_PRODUCTO as p 
on p.COD_PRODUCTO = ds.COD_PRODUCTO
where p.COD_CLASE = '00001' and ds.COD_PRODUCTO_DEV = ''
